419 {
420  if (d_data->first_free == address_of(node)) {
421  d_data->first_free = node->next_free;
422  if (d_data->first_free == detail::invalid_address) {
423  d_data->first_free = d_data->tail++;
424  if (d_data->first_free >= d_data->chunks)
425  d_data->first_free = detail::invalid_address;
426  else
427  //initialize tail
428  detail::initialize_chunk<T>(d_data->root_ptr() + d_data->first_free);
429  }
430  }
431 
432  //remove node from list of free chunks
433  if (node->prev_free != detail::invalid_address) {
434  //previous free chunk
435  detail::mem_pool_chunk<T>* prev = d_data->root_ptr() + node->prev_free;
436  prev->next_free = node->next_free; //set next free chunk
437  if (prev->next_free != detail::invalid_address)
438  //if next free chunk is valid, set its previous free chunk
439  (d_data->root_ptr() + prev->next_free)->prev_free = address_of(prev);

442  //next free chunk

453 T* contiguous_mem_pool<T, Alloc>::allocate()
454 {
455  if (d_data->first_free == detail::invalid_address) {
456  throw std::bad_alloc();// "contiguous_mem_pool full and cannot be resized");
457  }
458 
459  d_data->objects++;
460 
461  detail::mem_pool_chunk<T>* free_chunk = d_data->root_ptr() + d_data->first_free;
462  unsigned free_index = detail::first_free_element(free_chunk);
463 
464  if (free_chunk->frees == detail::all_free) {
465  d_data->used_chunks++;
466  //empty chunk, add it to the list of filled chunks
467  if (d_data->first_filled != detail::invalid_address)
468  {
469  detail::mem_pool_chunk<T>* filled_chunk = d_data->root_ptr() + d_data->first_filled;
470  filled_chunk->prev_filled = address_of(free_chunk);
471  free_chunk->next_filled = d_data->first_filled;
472  }
473  d_data->first_filled = address_of(free_chunk);
474  }
475 
476  //remove set bit
477  free_chunk->frees = free_chunk->frees & (~(1ULL << free_index));
478  if (free_chunk->frees == 0) {
479  //remove free_chunk from the linked list of free chunks
480  remove_free_node(free_chunk);
481  }
482 
483  return ((T*)free_chunk->elements + free_index);

533 template<class T, class Alloc>
534 bool contiguous_mem_pool<T, Alloc>::deallocate(T* ptr)
535 {
536  ASSERT_DEBUG(ptr && (void*)ptr >= (void*)d_data->root_ptr() && (void*)ptr < (void*)(d_data->root_ptr() + d_data->chunks),
537  "pointer does not belong to this pool");
538 
539  //retrieve the corresponding chunk
540  unsigned node_index = (unsigned)(((size_t)ptr - (size_t)d_data->root_ptr()) / sizeof(detail::mem_pool_chunk<T>));
541  detail::mem_pool_chunk<T>* node = d_data->root_ptr() + node_index;
542 
543  //sanity check
544  ASSERT_DEBUG(node_index < d_data->chunks, "Corrupted contiguous_mem_pool");
545 
546  //sanity check
547  unsigned index = (unsigned)(ptr - (T*)node->elements);
548 
549  //sanity check
550  ASSERT_DEBUG(((node->frees >> index) & 1) == 0, "Corrupted contiguous_mem_pool");
551 
552  if (node->frees == 0) {
553  //add it to the list of free chunks
554  add_to_free_chunks(node, node_index);
555  }
556 
557  //update frees mask
558  node->frees |= (1ULL << index);
559 
560  if (node->frees == detail::all_free) {
561  //this node is empty, remove it from the list of filled node
562  remove_filled_node(node);
563  d_data->used_chunks--;
564  }
565 
566  return d_data->objects-- == d_data->capacity;
567 }
