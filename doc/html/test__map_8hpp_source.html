511 template< class T>
512 int shellSort(T arr[], int n)
513 {
514  // Start with a big gap, then reduce the gap
515  for (int gap = n >> 1; gap > 0; gap >>= 1)
516  {
517  // Do a gapped insertion sort for this gap size.
518  // The first gap elements a[0..gap-1] are already in gapped order
519  // keep adding one more element until the entire array is
520  // gap sorted
521  for (int i = gap; i < n; i += 1)
522  {
523  // add a[i] to the elements that have been gap sorted
524  // save a[i] in temp and make a hole at position i
525  T temp = arr[i];
526 
527  // shift earlier gap-sorted elements up until the correct
528  // location for a[i] is found
529  int j;
530  for (j = i; j >= gap && arr[j - gap] > temp; j -= gap)
531  arr[j] = arr[j - gap];
532 
533  // put temp (the original a[i]) in its correct location
534  arr[j] = temp;
535  }
536  }
537  return 0;
538 }
