376  void rehash(size_t new_hash_mask, Iter first, Iter last)
377  {
378  // Rehash the table for given mask value.
379  // The old table is dropped before creating the new one, avoiding memory peaks.
380  // Do not check for potential duplicate values.
381 
382  try {
383  free_buckets(d_buckets);
384  d_buckets = null_node();
385  d_next_target = d_hash_mask = 0;
386  d_max_dist = 1;
387  d_buckets = make_buckets((new_hash_mask + 1ULL));
388  }
389  catch (...) {
390  // mark as dirty
391  mark_dirty();
392  throw;
393  }
394 
395  typename node_type::dist_type dist;
396  size_t hash, index;
397  size_t bsize = new_hash_mask + 1ULL;
398 
399  // Loop through values
400  for (auto it = first; it != last; ++it) {
401 
402  // Hash value
403  hash = hash_key(extract_key::key(*it));
404  // Compute index
405  index = (hash & new_hash_mask);
406 
407  if (SEQ_UNLIKELY(d_max_dist == node_type::max_distance)) {
408  //Pure linear hashing
409  while (!d_buckets[index].null() ) {
410  if (SEQ_UNLIKELY(++index == bsize))
411  index = 0;
412  }
413  // Insert
414  d_buckets[index] = node_type(node_type::small_hash(hash), index == (hash & new_hash_mask) ? 0 : node_type::max_distance, it);
415  continue;
416  }
417 
418  // Robin-hood probing
419  dist = 0;
420  while (dist <= d_buckets[index].distance()) {
421  if (SEQ_UNLIKELY(++index == bsize))
422  index = 0;
423  dist++;
424  }

647  void rehash(size_t size = 0, bool force = false)
648  {
649  // Rehash for given size if one of the following is true:
650  // - the hash table is dirty
651  // - the new hash table size is different from the current one
652  // - force is true
653 
654  const bool null_size = size == 0;
655 
656  if(null_size)
657  size = static_cast<size_t>(this->size() / static_cast<double>(d_load_factor));
658  if (size == 0)
659  return rehash(63, d_seq.end(), d_seq.end()); // Minimum size is 64
660  size_t new_hash_mask;
661  if ((size & (size - 1ULL)) == 0ULL) new_hash_mask = size - 1ULL;
662  else new_hash_mask = (1ULL << (1ULL + (bit_scan_reverse_64(size)))) - 1ULL;
663  if(dirty())
664  rehash_remove_duplicates(new_hash_mask, d_seq.begin(), d_seq.end());
665  else if (force || new_hash_mask != d_hash_mask || d_max_dist == node_type::max_distance) {
666  if (d_max_dist == node_type::max_distance && null_size) {
667  // linear hashing behavior: make sure to increase hash table size
668  new_hash_mask = ((new_hash_mask + 1U) * 2U) - 1U;
669  }
670  rehash(new_hash_mask, d_seq.begin(), d_seq.end());
671  }

758  auto insert_linear(Args&&... args) -> std::pair<iterator, bool>
759  {
760  // Insert value using linear probing
761

780  template<Location loc, class... Args >
781  auto insert(Args&&... args) -> std::pair<iterator, bool>
782  {
783  // Insert new key
784 
785  // Check for dirty
786  check_hash_operation();
787 
788  // Check for potential rehash.
789  // Avoid rehashing if the maximum distance is below 7.
790  if (SEQ_UNLIKELY(size() >= d_hash_mask || (d_max_dist > 7 && size() >= d_next_target) )) {
791  rehash(size() * 2U);
792  }
793 
794  // Check for purely linear hash table
795  if (SEQ_UNLIKELY(d_max_dist == node_type::max_distance))
796  return insert_linear<loc>(std::forward<Args>(args)...);
797 
798  // Build/extract value to insert (avoid building value on the stack if possible)
799  using store_type = typename BuildValue<Value, Args&&...>::type;
800  store_type value = BuildValue<Value, Args&&...>::build(std::forward<Args>(args)...);
801 
802  // Compute hash value
803  const size_t hash = hash_key(extract_key::key(value));
804  size_t index = (hash & d_hash_mask);
805  dist_type dist = 0;
806  tiny_hash h = node_type::small_hash(hash);
807 
808  // Find start insert location and check for equal value
809  while (dist <= d_buckets[index].distance()) {
810  if (SEQ_UNLIKELY(d_buckets[index].hash() == h && (*this)(extract_key::key(d_buckets[index].value()), extract_key::key(value))))
811  return std::pair< iterator, bool>(iterator(d_buckets[index].node(), d_buckets[index].pos()), false);
812  if (SEQ_UNLIKELY(++index == bucket_size()))
813  index = 0;
814  ++dist;
815  }
816 
817  // At this point, we are sure the key does not already exist in the table, let's proceed to the insertion
818 
819  // Update maximum distance
820  if (SEQ_UNLIKELY(dist > d_max_dist))
821  d_max_dist = dist;
822 
823  // Insert in sequence, might throw (fine)
824  iterator tmp_it = Inserter<sequence_type, loc, store_type>::insert(d_seq, value);
825 
826  // Move nodes based on distance (robin hood hashing), only if computed distance is not null
827  node_type n = d_buckets[index];
828  d_buckets[index] = node_type(h, dist, tmp_it);
829  if (dist )
830  start_insert(d_buckets,bucket_size(), index, dist, n);
831 
832  return std::pair< iterator, bool>(tmp_it,true);
833  }

902  auto erase_hash(size_t hash, const_iterator it) -> iterator
903  {
904  // Erase key
905 
906  SEQ_ASSERT_DEBUG(it != d_seq.end(), "invalid erase position");
907 
908  // Check for dirty
909  check_hash_operation();
910 
911  // Erase in sequence
912  auto res = d_seq.erase(it);
913 
914  // Find the node corresponding to this iterator
915  node_type* n = find_node(hash,it);
916  node_type* prev = n++;
917  const node_type* end = d_buckets + bucket_size();
918 
919  if (SEQ_UNLIKELY(n == end))
920  n = d_buckets;
921 
922  if (SEQ_UNLIKELY(d_max_dist == node_type::max_distance))
923  {
924  // Pure linear hash map: use tombstone
925  prev->empty_tombstone();
926  return res;
927  }
928 
929  // Robin hood backward shift deletion
930  dist_type dist = n->distance();
931  while (dist > 0) {
932  *prev = *n;
933  prev->set_distance(dist - 1);
934  prev = n++;
935  if (SEQ_UNLIKELY(n == end))
936  n = d_buckets;
937  dist = n->distance();
938  }
939  prev->empty();
940 
941  return res;
942  }
