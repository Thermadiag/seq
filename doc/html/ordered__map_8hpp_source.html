609  void rehash(size_t size = 0, bool force = false)
610  {
611  // Rehash for given size if one of the following is true:
612  // - the hash table is dirty
613  // - the new hash table size is different from the current one
614  // - force is true
615 
616  const bool null_size = size == 0;
617 
618  if(null_size)
619  size = static_cast<size_t>(this->size() / (double)d_load_factor);
620  if (size == 0)
621  return rehash(63, d_seq.end(), d_seq.end()); // Minimum size is 64
622  size_t new_hash_mask;
623  if ((size & (size - 1ULL)) == 0ULL) new_hash_mask = size - 1ULL;
624  else new_hash_mask = (1ULL << (1ULL + (bit_scan_reverse_64(size)))) - 1ULL;
625  if(dirty())
626  rehash_remove_duplicates(new_hash_mask, d_seq.begin(), d_seq.end());
627  else if (force || new_hash_mask != d_hash_mask || d_max_dist == node_type::max_distance) {
628  if (d_max_dist == node_type::max_distance && null_size) {
629  // linear hashing behavior: make sure to increase hash table size
630  new_hash_mask = ((new_hash_mask + 1U) * 2U) - 1U;
631  }
632  rehash(new_hash_mask, d_seq.begin(), d_seq.end());
633  }

742  template<Location loc, class... Args >
743  auto insert(Args&&... args) -> std::pair<iterator, bool>
744  {
745  // Insert new key
746 
747  // Check for dirty
748  check_hash_operation();
749 
750  // Check for potential rehash.
751  // Avoid rehashing if the maximum distance is below 7.
752  if (SEQ_UNLIKELY(size() >= d_hash_mask || (d_max_dist > 7 && size() >= d_next_target) )) {
753  rehash(size() * 2U);
754  }
755 
756  // Check for purely linear hash table
757  if (SEQ_UNLIKELY(d_max_dist == node_type::max_distance))
758  return insert_linear<loc>(std::forward<Args>(args)...);
759 
760  // Build/extract value to insert (avoid building value on the stack if possible)
761  using store_type = typename BuildValue<Value, Args&&...>::type;
762  store_type value = BuildValue<Value, Args&&...>::build(std::forward<Args>(args)...);
763 
764  // Compute hash value
765  const size_t hash = hash_key(extract_key::key(value));
766  size_t index = (hash & d_hash_mask);
767  dist_type dist = 0;
768  tiny_hash h = node_type::small_hash(hash);
769 
770  // Find start insert location and check for equal value
771  while (dist <= d_buckets[index].distance()) {
772  if (SEQ_UNLIKELY(d_buckets[index].hash() == h && (*this)(extract_key::key(d_buckets[index].value()), extract_key::key(value))))
773  return std::pair< iterator, bool>(iterator(d_buckets[index].node(), d_buckets[index].pos()), false);
774  if (SEQ_UNLIKELY(++index == bucket_size()))
775  index = 0;
776  ++dist;
777  }
778 
779  // At this point, we are sure the key does not already exist in the table, let's proceed to the insertion
780 
781  // Update maximum distance
782  if (SEQ_UNLIKELY(dist > d_max_dist))
783  d_max_dist = dist;
784 
785  // Insert in sequence
786  iterator tmp_it = Inserter<sequence_type, loc, store_type>::insert(d_seq, value);
787 
788  // Move nodes based on distance (robin hood hashing), only if computed distance is not null
789  node_type n = d_buckets[index];
790  d_buckets[index] = node_type(h, dist, tmp_it);
791  if (dist )
792  start_insert(d_buckets,bucket_size(), index, dist, n);
793 
794  return std::pair< iterator, bool>(tmp_it,true);
795  }

864  auto erase_hash(size_t hash, const_iterator it) -> iterator
865  {
866  // Erase key
867 
868  SEQ_ASSERT_DEBUG(it != d_seq.end(), "invalid erase position");
869 
870  // Check for dirty
871  check_hash_operation();
872 
873  // Find the node corresponding to this iterator
874  node_type* n = find_node(hash,it);
875  node_type* prev = n++;
876  const node_type* end = d_buckets + bucket_size();
877 
878  if (SEQ_UNLIKELY(n == end))
879  n = d_buckets;
880 
881  if (d_max_dist == node_type::max_distance)
882  {
883  // Pure linear hash map: use tombstone
884  prev->empty_tombstone();
885  }
886  else
887  {
888  // Robin hood backward shift deletion
889  dist_type dist = n->distance();
890  while (dist > 0) {
891  *prev = *n;
892  prev->set_distance(dist - 1);
893  prev = n++;
894  if (SEQ_UNLIKELY(n == end))
895  n = d_buckets;
896  dist = n->distance();
897  }
898  prev->empty();
899  }
900 
901  // Erase in sequence
902  return d_seq.erase(it);
903  }
