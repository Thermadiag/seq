616  void rehash(size_t size = 0, bool force = false)
617  {
618  // Rehash for given size if one of the following is true:
619  // - the hash table is dirty
620  // - the new hash table size is different from the current one
621  // - force is true
622 
623  const bool null_size = size == 0;
624 
625  if(null_size)
626  size = static_cast<size_t>(this->size() / (double)d_load_factor);
627  if (size == 0)
628  return rehash(63, d_seq.end(), d_seq.end()); // Minimum size is 64
629  size_t new_hash_mask;
630  if ((size & (size - 1ULL)) == 0ULL) new_hash_mask = size - 1ULL;
631  else new_hash_mask = (1ULL << (1ULL + (bit_scan_reverse_64(size)))) - 1ULL;
632  if(dirty())
633  rehash_remove_duplicates(new_hash_mask, d_seq.begin(), d_seq.end());
634  else if (force || new_hash_mask != d_hash_mask || d_max_dist == node_type::max_distance) {
635  if (d_max_dist == node_type::max_distance && null_size) {
636  // linear hashing behavior: make sure to increase hash table size
637  new_hash_mask = ((new_hash_mask + 1U) * 2U) - 1U;
638  }
639  rehash(new_hash_mask, d_seq.begin(), d_seq.end());

749  template<Location loc, class... Args >
750  std::pair<iterator,bool> insert(Args&&... args)
751  {
752  // Insert new key
753 
754  // Check for dirty
755  check_hash_operation();
756 
757  // Check for potential rehash.
758  // Avoid rehashing if the maximum distance is below 7.
759  if (SEQ_UNLIKELY(size() >= d_hash_mask || (d_max_dist > 7 && size() >= d_next_target) )) {
760  rehash(size() * 2U);
761  }
762 
763  // Check for purely linear hash table
764  if (SEQ_UNLIKELY(d_max_dist == node_type::max_distance))
765  return insert_linear<loc>(std::forward<Args>(args)...);
766 
767  // Build/extract value to insert (avoid building value on the stack if possible)
768  using store_type = typename BuildValue<Value, Args&&...>::type;
769  store_type value = BuildValue<Value, Args&&...>::build(std::forward<Args>(args)...);
770 
771  // Compute hash value
772  const size_t hash = hash_key(extract_key::key(value));
773  size_t index = (hash & d_hash_mask);
774  dist_type dist = 0;
775  tiny_hash h = node_type::small_hash(hash);
776 
777  // Find start insert location and check for equal value
778  while (dist <= d_buckets[index].distance()) {
779  if (SEQ_UNLIKELY(d_buckets[index].hash() == h && (*this)(extract_key::key(d_buckets[index].value()), extract_key::key(value))))
780  return std::pair< iterator, bool>(iterator(d_buckets[index].node(), d_buckets[index].pos()), false);
781  if (SEQ_UNLIKELY(++index == bucket_size()))
782  index = 0;
783  ++dist;
784  }
785 
786  // At this point, we are sure the key does not already exist in the table, let's proceed to the insertion
787 
788  // Update maximum distance
789  if (SEQ_UNLIKELY(dist > d_max_dist))
790  d_max_dist = dist;
791 
792  // Insert in sequence
793  iterator tmp_it = Inserter<sequence_type, loc, store_type>::insert(d_seq, value);
794 
795  // Move nodes based on distance (robin hood hashing), only if computed distance is not null
796  node_type n = d_buckets[index];
797  d_buckets[index] = node_type(h, dist, tmp_it);
798  if (dist )

800

871  iterator erase_hash(size_t hash, const_iterator it)
872  {
873  // Erase key
874 
875  SEQ_ASSERT_DEBUG(it != d_seq.end(), "invalid erase position");
876 
877  // Check for dirty
878  check_hash_operation();
879 
880  // Find the node corresponding to this iterator
881  node_type* n = find_node(hash,it);
882  node_type* prev = n++;
883  const node_type* end = d_buckets + bucket_size();
884 
885  if (SEQ_UNLIKELY(n == end))
886  n = d_buckets;
887 
888  if (d_max_dist == node_type::max_distance)
889  {
890  // Pure linear hash map: use tombstone
891  prev->empty_tombstone();
892  }
893  else
894  {
895  // Robin hood backward shift deletion
896  dist_type dist = n->distance();
897  while (dist > 0) {
898  *prev = *n;
899  prev->set_distance(dist - 1);
900  prev = n++;
901  if (SEQ_UNLIKELY(n == end))
902  n = d_buckets;
903  dist = n->distance();
904  }
905  prev->empty();
906  }
907 
908  // Erase in sequence
909  return d_seq.erase(it);
910  }
