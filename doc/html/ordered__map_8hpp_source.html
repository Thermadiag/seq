618  tiny_hash h = node_type::small_hash(hash);
619  index = mask_hash(hash , new_hash_mask, new_hash_len);
620 
621  dist = 0;
622  while (!d_buckets[index].null() && static_cast<difference_type>(dist) <= static_cast<difference_type>(d_buckets[index].distance())) {
623  if (h == d_buckets[index].hash() && (*this)(extract_key::key(d_buckets[index].value()), extract_key::key (*it)))
624  {
625  dist = static_cast<size_t>(-1);
626  break;
627  }
628  if (SEQ_UNLIKELY(++index == bsize))
629  index = 0;
630  dist++;
631  }
632

635  while (!d_buckets[index].null()) {

666  ++it;
667  }

798  void rehash(size_t size = 0, bool force = false)
799  {
800  // Rehash for given size if one of the following is true:
801  // - the hash table is dirty
802  // - the new hash table size is different from the current one
803  // - force is true
804 
805  const bool null_size = (size == 0);
806 
807  if(null_size)
808  size = static_cast<size_t>(static_cast<double>(this->size()) / static_cast<double>(d_load_factor));
809  if (size == 0)
810  return rehash(63,6, d_seq.end(), d_seq.end()); // Minimum size is 64
811  size_t new_hash_mask;
812  size_t new_hash_len;
813  if ((size & (size - 1ULL)) == 0ULL) new_hash_mask = size - 1ULL;
814  else new_hash_mask = (1ULL << (1ULL + (bit_scan_reverse_64(size)))) - 1ULL;
815  new_hash_len = bit_scan_reverse_64(new_hash_mask) + 1U;
816  if(dirty())
817  rehash_remove_duplicates(new_hash_mask, new_hash_len, d_seq.begin(), d_seq.end());
818  else if (force || new_hash_mask != d_hash_mask || d_max_dist == node_type::max_distance) {
819  if (d_max_dist == node_type::max_distance && null_size) {
820  // linear hashing behavior: make sure to increase hash table size
821  new_hash_mask = ((new_hash_mask + 1U) * 2U) - 1U;
822  }
823  rehash(new_hash_mask, new_hash_len, d_seq.begin(), d_seq.end());
824  }
