568  SEQ_ASSERT_DEBUG(to <= str.size(), "");

570  size_t f_size = to - from;
571  if (f_size == w.width || w.alignment == width_format::align_none)
572  return;
573 
574  if (f_size > w.width)
575  {
576  size_t diff = (f_size - w.width);
577  //shrink string
578  if (w.alignment == width_format::align_right) {
579  memmove(const_cast<char*>(str.data()) + from, str.data() + to - w.width, str.size() - to + w.width);
580  }
581  else if (w.alignment == width_format::align_left) {
582  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, str.size() - to);
583  }
584  else {
585  memmove(const_cast<char*>(str.data()) + from, str.data() + from + diff / 2, w.width);
586  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, str.size() - to);
587  }
588  str.resize(str.size() - diff);
589  }
590  else
591  {
592  //enlarge string
593  size_t old_size = str.size();

595 
596  //move the right part
597  if (to != str.size()) {
598  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, old_size - to);
599  }
600
