579  SEQ_ASSERT_DEBUG(to <= str.size(), "");

582  if (f_size == w.width || w.alignment == width_format::align_none)
583  return;
584 
585  if (f_size > w.width)
586  {
587  size_t diff = (f_size - w.width);
588  //shrink string
589  if (w.alignment == width_format::align_right) {
590  memmove(const_cast<char*>(str.data()) + from, str.data() + to - w.width, str.size() - to + w.width);
591  }
592  else if (w.alignment == width_format::align_left) {
593  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, str.size() - to);
594  }
595  else {
596  memmove(const_cast<char*>(str.data()) + from, str.data() + from + diff / 2, w.width);
597  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, str.size() - to);
598  }
599  str.resize(str.size() - diff);
600  }
601  else
602  {
603  //enlarge string
604  size_t old_size = str.size();
605  str.resize(str.size() - f_size + w.width);
606 
607  //move the right part
608  if (to != str.size()) {
609  memmove(const_cast<char*>(str.data()) + from + w.width, str.data() + to, old_size - to);
610  }
611 
612  if (w.alignment == width_format::align_right) {
613  memmove(const_cast<char*>(str.data()) + from + w.width - f_size, str.data() + from, f_size);
614  memset(const_cast<char*>(str.data()) + from, w.pad, w.width - f_size);
