903  {
904  // Sort again the tiered_vector, but only if:
905  // - the flat_tree is dirty (and potentially unsorted),
906  // - and the tiered_vector is not already sorted.
907 
908  if (d_deque.empty() || d_dirty == 0)
909  return;

911  iterator it = d_deque.begin();

917  iterator prev = it++;
918  while (it != en && sorted) {
919  if ((*this)(*it, *prev))
920  sorted = false;
921  else if (Unique && !(*this)(*prev, *it))
922  unique = false;
923  ++it;

929  sort_deque<Stable>(d_deque, 0, d_deque.size(), Less(this));

931  iterator it = std::unique(d_deque.begin(), d_deque.end(), Equal(this));
932  d_deque.erase(it, d_deque.end());
933  }
