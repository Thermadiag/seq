15 static inline unsigned INC_common_bytes(size_t val)
16 {
17 #if VIP_BYTE_ORDER == VIP_LITTLE_ENDIAN
18  if (sizeof(val) == 8) {

21  _BitScanForward64(&r, (uint64_t)val);
22  return (int)(r >> 3);
23 # elif (defined(__clang__) || (defined(__GNUC__) && (__GNUC__>=3))) && !defined(LZ4_FORCE_SW_BITCOUNT)
24  return (__builtin_ctzll((uint64_t)val) >> 3);
25 # else
26  static const int DeBruijnBytePos[64] = { 0, 0, 0, 0, 0, 1, 1, 2,
27  0, 3, 1, 3, 1, 4, 2, 7,
28  0, 2, 3, 6, 1, 5, 3, 5,
29  1, 3, 4, 4, 2, 5, 6, 7,
30  7, 0, 1, 2, 3, 3, 4, 6,
31  2, 6, 5, 5, 3, 4, 5, 6,
32  7, 1, 2, 4, 6, 4, 4, 5,
33  7, 2, 6, 5, 7, 6, 7, 7 };
34  return DeBruijnBytePos[((uint64_t)((val & -(long long)val) * 0x0218A392CDABBD3FULL)) >> 58];
35 # endif
36  }
37  else /* 32 bits */ {
38 # if defined(_MSC_VER) && !defined(LZ4_FORCE_SW_BITCOUNT)
39  unsigned long r;
40  _BitScanForward(&r, (uint32_t)val);
41  return (int)(r >> 3);
42 # elif (defined(__clang__) || (defined(__GNUC__) && (__GNUC__>=3))) && !defined(LZ4_FORCE_SW_BITCOUNT)
43  return (__builtin_ctz((uint32_t)val) >> 3);
44 # else
45  static const int DeBruijnBytePos[32] = { 0, 0, 3, 0, 3, 1, 3, 0,
46  3, 2, 2, 1, 3, 2, 0, 1,
47  3, 3, 1, 2, 2, 2, 2, 0,
48  3, 1, 2, 0, 1, 0, 1, 1 };
49  return DeBruijnBytePos[((uint32_t)((val & -(S32)val) * 0x077CB531U)) >> 27];
50 # endif
51  }
52 #else
53  /* Big Endian CPU */
54  if (sizeof(val) == 8) { /* 64-bits */
55 # if defined(_MSC_VER) && defined(_WIN64) && !defined(LZ4_FORCE_SW_BITCOUNT)

57  _BitScanReverse64(&r, val);

63  Just to avoid some static analyzer complaining about shift by 32 on 32-bits target.
64  Note that this code path is never triggered in 32-bits mode. */
65  unsigned r;
66  if (!(val >> by32)) { r = 4; }
67  else { r = 0; val >>= by32; }
68  if (!(val >> 16)) { r += 2; val >>= 8; }
69  else { val >>= 24; }
70  r += (!val);
71  return r;
72 # endif
73  }
74  else /* 32 bits */ {

77  _BitScanReverse(&r, (unsigned long)val);
78  return (unsigned)(r >> 3);

81 # else
82  unsigned r;
83  if (!(val >> 16)) { r = 2; val >>= 8; }
