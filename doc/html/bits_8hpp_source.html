405  inline auto handmade_aligned_malloc(size_t size, size_t alignment) -> void*
406  {
407  void* ptr = nullptr;
408  alignment--;
409 
410  size_t offset = 0;
411  std::uint8_t* mem = nullptr;
412 
413  // Room for padding and extra pointer stored in front of allocated area
414  size_t overhead = alignment + sizeof(void*);
415 
416  // Avoid integer overflow
417  if (size > (SIZE_MAX - overhead)) {
418  return nullptr;
419  }
420 
421  mem = static_cast<std::uint8_t*>(malloc(size + overhead));
422  if (mem == nullptr) {
423  return mem;
424 }
425 
426  // Use the fact that alignment + 1U is a power of 2
427  offset = ((alignment ^ (reinterpret_cast<std::uintptr_t>(mem + sizeof(void*)) & alignment)) + 1U) & alignment;
428  ptr = static_cast<void*>(mem + sizeof(void*) + offset);
429  (static_cast<void**>(ptr))[-1] = mem;
430  return ptr;
431  }

544  // Define default popcount functions
545 
546  // This uses fewer arithmetic operations than any other known
547  // implementation on machines with fast multiplication.
548  // It uses 12 arithmetic operations, one of which is a multiply.
549  // http://en.wikipedia.org/wiki/Hamming_weight#Efficient_implementation
550  inline auto popcount64(std::uint64_t x) -> unsigned
551  {
552  std::uint64_t m1 = 0x5555555555555555LL;
553  std::uint64_t m2 = 0x3333333333333333LL;
554  std::uint64_t m4 = 0x0F0F0F0F0F0F0F0FLL;
555  std::uint64_t h01 = 0x0101010101010101LL;
556 
557  x -= (x >> 1) & m1;
558  x = (x & m2) + ((x >> 2) & m2);
559  x = (x + (x >> 4)) & m4;
560 
561  return (x * h01) >> 56;
562  }
563  inline auto popcount32(uint32_t i) -> unsigned
564  {
565  i = i - ((i >> 1) & 0x55555555); // add pairs of bits
566  i = (i & 0x33333333) + ((i >> 2) & 0x33333333); // quads
567  i = (i + (i >> 4)) & 0x0F0F0F0F; // groups of 8
568  return (i * 0x01010101) >> 24; // horizontal sum of bytes
569  }
570  }
