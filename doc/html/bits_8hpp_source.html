335  inline void* handmade_aligned_malloc(size_t size, size_t alignment)
336  {
337  void* ptr;
338  alignment--;
339 
340  size_t offset;
341  std::uint8_t* mem;
342 
343  // Room for padding and extra pointer stored in front of allocated area
344  size_t overhead = alignment + sizeof(void*);
345 
346  // Avoid integer overflow
347  if (size > (SIZE_MAX - overhead)) {
348  return NULL;
349  }
350 
351  mem = static_cast<std::uint8_t*>(malloc(size + overhead));
352  if (mem == NULL)
353  return mem;
354 
355  // Use the fact that alignment + 1U is a power of 2
356  offset = ((alignment ^ (reinterpret_cast<std::uintptr_t>(mem + sizeof(void*)) & alignment)) + 1U) & alignment;
357  ptr = static_cast<void*>(mem + sizeof(void*) + offset);
358  ((void**)ptr)[-1] = mem;
359  return ptr;
360  }

473  // Define default popcount functions
474 
475  // This uses fewer arithmetic operations than any other known
476  // implementation on machines with fast multiplication.
477  // It uses 12 arithmetic operations, one of which is a multiply.
478  // http://en.wikipedia.org/wiki/Hamming_weight#Efficient_implementation
479  inline unsigned popcount64(std::uint64_t x)
480  {
481  std::uint64_t m1 = 0x5555555555555555ll;
482  std::uint64_t m2 = 0x3333333333333333ll;
483  std::uint64_t m4 = 0x0F0F0F0F0F0F0F0Fll;
484  std::uint64_t h01 = 0x0101010101010101ll;
485 
486  x -= (x >> 1) & m1;
487  x = (x & m2) + ((x >> 2) & m2);
488  x = (x + (x >> 4)) & m4;
489 
490  return (x * h01) >> 56;
491  }
492  inline unsigned popcount32(uint32_t i)
493  {
494  i = i - ((i >> 1) & 0x55555555); // add pairs of bits
495  i = (i & 0x33333333) + ((i >> 2) & 0x33333333); // quads
496  i = (i + (i >> 4)) & 0x0F0F0F0F; // groups of 8
497  return (i * 0x01010101) >> 24; // horizontal sum of bytes
498  }
499  }
