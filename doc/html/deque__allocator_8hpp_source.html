122  auto rebind_for(size_t elems_per_chunk) -> bool
123  {
124  size_t required_T = elems_per_chunk;
125  if (required_T > total_T)
126  return false;
127  count = total_T / required_T;
128  this->elems_per_chunk = elems_per_chunk;
129  //create chain of free
130  set_next(chunks, NULL);
131  first_free = NULL;
132  tail = 0;
133  used = 0;
134  return true;

143  auto allocate() noexcept -> T* {
144  if (first_free) {

245  //allocate one chunk. returns the usable part as a T*

247  {
248  if (capacity == objects) {
