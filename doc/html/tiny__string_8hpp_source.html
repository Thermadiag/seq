102  # else
103  static const int DeBruijnBytePos[64] = { 0, 0, 0, 0, 0, 1, 1, 2,
104  0, 3, 1, 3, 1, 4, 2, 7,
105  0, 2, 3, 6, 1, 5, 3, 5,

131  # if defined(_MSC_VER) && defined(_WIN64)
