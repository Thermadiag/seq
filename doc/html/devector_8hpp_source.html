117  void copy_destroy_input( T* first, T* last, T* dst) {
118  // Copy range [first, last) to non overlapping dst, and destroy input range.
119  // In case of exception, input is not detroyed, and created values are destroyed
120  // Strong exception guarantee

122  memcpy(static_cast<void*>(dst), static_cast<void*>(first), static_cast<size_t>(last - first) * sizeof(T));
