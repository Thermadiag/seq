739  goto partial;

742  first = str.getc();
743  if (!is_digit(first))

745  }
746  else if (!is_digit(first))

793  template<class T, class Stream>
794  SEQ_ALWAYS_INLINE T read_double(Stream& str, chars_format fmt, char dot = '.')
795  {
796  using Integral = typename Stream::streamsize;
797 
798  str.reset();
799  Integral saved = str.tell();
800  int first = str.getc();
801  int sign = 1, c;
802  T res = 0;
803 
804 
805  // skip spaces
806  while (is_space(first)) {
807  first = str.getc();
808  }
809  // check EOF
810  if (SEQ_UNLIKELY(first == EOF))
811  return (T)0;
812 
813  if (first == '-') {
814  // first character is a sign
815  sign = -1;
816  first = str.getc();
817  if (SEQ_UNLIKELY(first == EOF))
818  return (T)0;
819  }
820  else if (first == '+') {
821  // first character is a sign
822  first = str.getc();
823  if (SEQ_UNLIKELY(first == EOF))
824  return (T)0;
825  }
826  // potential nan
827  else if (SEQ_UNLIKELY(to_upper(first) == 'N')) {
828  // read 'nan'
829  return read_nan<T>(saved, str);
830  }
831  // read potential infinity
832  if (SEQ_UNLIKELY(first == 'i' || first == 'I')) {
833  // read 'inf'
834  return read_inf<T>(saved,sign, str);
835  }
836  else if (SEQ_UNLIKELY(!is_digit(first) && first != dot))
837  // first character must be either the dot or a digit
838  {
839  return return_bad_format<T>(saved, str);
840  }
841 
842 
843  // fast path: read integral part and decimal part as integers
844  str.back();
845  Integral check_point = str.tell();
846  // read intergal part
847  int64_t integral = read_int64(str);
848  if (SEQ_UNLIKELY(!str)) {
849  if (integral == -1)
850  //too big: switch to the default slower version
851  goto slow_path;
852 
853  // invalid intergal part: invalid format
854  return return_bad_format<T>(saved, str);
855  }
856 
857  // keep going with dot, decimal part, exponent.
858  // mark this state as a chack point, as the intergal part might be reused by the slow path.
859  res = (T)integral;
860  check_point = str.tell();
861 
862  c = str.getc();
863  if (c == dot) {
864  //read decimal part
865  Integral new_check_point = str.tell();
866  int64_t decimal = read_int64(str);
867  if (SEQ_UNLIKELY(!str)) {
868  if (decimal == -1) {
869  //too big: switch to the default slower version
870  goto slow_path;
871  }
872  if (fmt == scientific)
873  // exponent is mandatory
874  return return_bad_format<T>(saved, str);
875  //we accept a dot with nothing after
876  return sign_value(res, sign);
877  }
878  int dist = (int)(str.tell() - new_check_point);
879  res += decimal * get_pow<T>(-dist);
880  c = str.getc();
881  }
882  if (c == 'e' || c == 'E') {
883  if (SEQ_UNLIKELY(fmt == fixed)) {
884  // exponent forbiden
885  return return_bad_format<T>(saved, str);
886  }
887  //read exponent
888  int exp = 0;
889  int esign = 1;
890  c = str.getc();
891  if (!is_digit(c)) {
892  //read exponent sign
893  if (c == '-') esign = -1;
894  else if (SEQ_UNLIKELY(c != '+'))
895  return return_bad_format<T>(saved, str);
896  c = str.getc();
897  if (!is_digit(c))
898  return return_bad_format<T>(saved, str);
899  }
900  // read exponent value
901  unsigned cc = c- '0';
902  do {
903  exp = (exp * 10) + (cc);
904  cc = str.getc() - '0';
905  } while (cc <= 9);
906  if (cc != _EOF)
907  str.back();
908  else
909  str.reset();
910 
911  exp = sign_value((int)exp, esign);
912  // check exponent validity
913  if (SEQ_UNLIKELY(exp > std::numeric_limits<T>::max_exponent10))
914  return sign_value(std::numeric_limits<T>::infinity(), sign);
915  else if (SEQ_UNLIKELY(exp < std::numeric_limits<T>::min_exponent10))
916  return sign_value((T)0, sign);
917 
918  // success
919  return sign_value(res * get_pow<T>(exp), sign);
920  }
921  else if (SEQ_UNLIKELY(fmt == scientific)) {
922  // exponent is mandatory
923  return return_bad_format<T>(saved, str);
924  }
925  else {
926  // success
927  str.reset();
928  if (c != EOF)
929  str.back();
930  return sign_value(res, sign);
931  }
932 
933  slow_path:
934  // slower way of reading a floating point value, relies on floating point arithmetic.
935  // reuse the integral part if possible.
936  str.seek(check_point);
937  return read_long_double<T>(integral != -1 ? res : std::numeric_limits<T>::quiet_NaN(), saved, sign, dot, str,fmt);
938  }

1386  {

1388  // Uses width to compute the proper exponent and decimal value.
1389

1392

1399  width = width - 1 - exponent;
1400  }
1401  else{
1402  fmt.fmt = seq::scientific;
1403  width = width - 1;
1404  }
1405  }

1408  exponent = 0;
1409  }

1413
