798  template<class T, class Stream>
799  SEQ_ALWAYS_INLINE auto read_double(Stream& str, chars_format fmt, char dot) -> T
800  {
801  using Integral = typename Stream::streamsize;
802 
803  str.reset();
804  Integral saved = str.tell();
805  int first = str.getc();
806  int sign = 1;
807  int c;
808  T res = 0;
809 
810 
811  // skip spaces
812  while (is_space(first)) {
813  first = str.getc();
814  }
815  // check EOF
816  if (SEQ_UNLIKELY(first == EOF)) {
817  return (T)0;
818 }
819 
820  if (first == '-') {
821  // first character is a sign
822  sign = -1;
823  first = str.getc();
824  if (SEQ_UNLIKELY(first == EOF)) {
825  return (T)0;
826 }
827  }
828  else if (first == '+') {
829  // first character is a sign
830  first = str.getc();
831  if (SEQ_UNLIKELY(first == EOF)) {
832  return (T)0;
833 }
834  }
835  // potential nan
836  else if (SEQ_UNLIKELY(to_upper(first) == 'N')) {
837  // read 'nan'
838  return read_nan<T>(saved, str);
839  }
840  // read potential infinity
841  if (SEQ_UNLIKELY(first == 'i' || first == 'I')) {
842  // read 'inf'
843  return read_inf<T>(saved,sign, str);
844  }
845  if (SEQ_UNLIKELY(!is_digit(first) && first != dot))
846  // first character must be either the dot or a digit
847  {
848  return return_bad_format<T>(saved, str);
849  }
850 
851 
852  // fast path: read integral part and decimal part as integers
853  str.back();
854  Integral check_point = str.tell();
855  // read intergal part
856  int64_t integral = read_int64(str);
857  if (SEQ_UNLIKELY(!str)) {
858  if (integral == -1) {
859  //too big: switch to the default slower version
860  goto slow_path;
861 }
862 
863  // invalid intergal part: invalid format
864  return return_bad_format<T>(saved, str);
865  }
866 
867  // keep going with dot, decimal part, exponent.
868  // mark this state as a chack point, as the intergal part might be reused by the slow path.
869  res = (T)integral;
870  check_point = str.tell();
871 
872  c = str.getc();
873  if (c == dot) {
874  //read decimal part
875  Integral new_check_point = str.tell();
876  int64_t decimal = read_int64(str);
877  if (SEQ_UNLIKELY(!str)) {
878  if (decimal == -1) {
879  //too big: switch to the default slower version
880  goto slow_path;
881  }
882  if (fmt == scientific) {
883  // exponent is mandatory
884  return return_bad_format<T>(saved, str);
885 }
886  //we accept a dot with nothing after
887  return sign_value(res, sign);
888  }
889  int dist = (int)(str.tell() - new_check_point);
890  res += decimal * get_pow<T>(-dist);
891  c = str.getc();
892  }
893  if (c == 'e' || c == 'E') {
894  if (SEQ_UNLIKELY(fmt == fixed)) {
895  // exponent forbiden
896  return return_bad_format<T>(saved, str);
897  }
898  //read exponent
899  int exp = 0;
900  int esign = 1;
901  c = str.getc();
902  if (!is_digit(c)) {
903  //read exponent sign
904  if (c == '-') { esign = -1;
905  } else if (SEQ_UNLIKELY(c != '+')) {
906  return return_bad_format<T>(saved, str);
907 }
908  c = str.getc();
909  if (!is_digit(c)) {
910  return return_bad_format<T>(saved, str);
911 }
912  }
913  // read exponent value
914  unsigned cc = c- '0';
915  do {
916  exp = (exp * 10) + (cc);
917  cc = str.getc() - '0';
918  } while (cc <= 9);
919  if (cc != _EOF) {
920  str.back();
921  } else {
922  str.reset();
923 }
924 
925  exp = sign_value(exp, esign);
926  // check exponent validity
927  if (SEQ_UNLIKELY(exp > std::numeric_limits<T>::max_exponent10)) {
928  return sign_value(std::numeric_limits<T>::infinity(), sign);
929  } if (SEQ_UNLIKELY(exp < std::numeric_limits<T>::min_exponent10))
930  return sign_value((T)0, sign);
931 
932  // success
933  T p = get_pow<T>(exp);
934  res = sign_value(res * p, sign);
935  return res;
936  }
937  if (SEQ_UNLIKELY(fmt == scientific)) {
938  // exponent is mandatory
939  return return_bad_format<T>(saved, str);
940  }
941  else {
942  // success
943  str.reset();
944  if (c != EOF)
945  str.back();
946  return sign_value(res, sign);
947  }
948 
949  slow_path:
950  // slower way of reading a floating point value, relies on floating point arithmetic.
951  // reuse the integral part if possible.
952  str.seek(check_point);
953  return read_long_double<T>(integral != -1 ? res : std::numeric_limits<T>::quiet_NaN(), saved, sign, dot, str,fmt);
954  }

1408  template<class T, class UInt>
1409  SEQ_ALWAYS_INLINE auto split_double(T value, UInt& integral, UInt& decimals, int16_t& exponent, bool & null_first, int width, float_chars_format& fmt) -> int
1410  {
1411  // Split floating point value into its integral, decimal and exponent part.
1412  // Uses width to compute the proper exponent and decimal value.
1413 
1414  static constexpr int max_exp_for_fixed = sizeof(T) <= 4 ? 8 : 17;
1415 
1416  T saved = value;
1417  exponent = normalize_double(value, fmt);
1418 
1419  if (fmt.fmt == seq::general) {
1420  //see https://stackoverflow.com/questions/30658919/the-precision-of-printf-with-specifier-g
1421  width = width == 0 ? 1 : width;
1422  // adjust format and precision
1423  if (width > exponent && exponent >= -4 && exponent <= max_exp_for_fixed){
1424  fmt.fmt = seq::fixed;
1425  width = width - 1 - exponent;
1426  }
1427  else{
1428  fmt.fmt = seq::scientific;
1429  width = width - 1;
1430  }
1431  }
1432  if (fmt.fmt == seq::fixed) {
1433  value = saved;
1434  exponent = 0;
1435  }
1436 
1437  integral = (UInt)value;
1438  T remainder = value - integral;
1439 
1440  null_first = false;
1441  if (remainder != 0 && remainder < 0.1) {
1442  // remainder is inf to 0.1, we must add 0.1 to avoid truncating the leading zeros
1443  remainder += 0.1;
1444  null_first = true;
1445  }
1446 
1447 
1448  int p = width > 17 ? 17 : width;
1449  remainder *= get_pow<T>(p);
1450  decimals = (UInt)remainder;
1451 
1452  // rounding
1453  remainder -= decimals;
1454  if (remainder >= 0.5) {
1455  decimals++;
1456  const UInt max_val = static_cast<UInt>(get_pow<T>(p) + (T)0.5);
1457  if (SEQ_UNLIKELY(decimals >= max_val)) {
1458  decimals = 0;
1459  integral++;
1460  if (exponent != 0 && integral >= 10) {
1461  exponent++;
1462  integral = 1;
1463  }
1464  }
1465  }
1466 
1467  return width;
1468  }
